CREATE VIEW BookingSummaryView AS
SELECT 
    rs.ReservationID,
    u.UserName,
    u.E_mail,
    u.Address,
    u.Phone_Num,
    rs.StartTime,
    rs.EndTime,
    rs.TotalPrice,
    rs.Status,
    rs.Purpose,
    rs.NumberGuest,
    rs.UserComment,
    rs.AdminComment,
    rs.ForeheadVisitTime,
    r.RoomName,
    r.Capacity
FROM 
    Reservation rs
INNER JOIN 
    [User] u ON rs.UserID = u.UserID
INNER JOIN 
    Room r ON rs.RoomID = r.RoomID;
